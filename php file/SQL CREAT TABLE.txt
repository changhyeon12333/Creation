CREATE TABLE SHOT_INFO(
    LATITUDE FLOAT NOT NULL,
    LONGITUDE FLOAT NOT NULL,
    PRIMARY KEY(LATITUDE,LONGITUDE)
    )

CREATE TABLE DRIVING (
    CARNUM VARCHAR NOT NULL,
    LECORD_TIME TIMESTAMP NOT NULL,
    ACEL FLOAT,
    BREAK FLOAT,
    STOP_SHORT BOOLEAN,
    
    FOREIGN KEY(CARNUM) REFERENCES USER(userCarnum),
    PRIMARY KEY(CARNUM , LECORD_TIME)
    )
